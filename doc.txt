| GETTING STARTED

   # composer create-project laravel/laravel project-name
   # composer require laravel/ui  
   # php artisan ui bootstrap --auth  =>  This command create all routes, controllers and views That we need for authentication
   # npm install 
   # npm run dev 

   # Creating migrations files ( Database Tables Structure )  =>  php artisan make:migration create_users_table
   # Creating The Database Tables                             =>  php artisan migrate

   # Creting The Route                                        =>  Route::get('/', [usersController::class, 'index'])->name('users')
   
   # Creating The Controller                                  =>  php artisan make:controller UsersController
   # Creating The Model                                       =>  php artisan make:model User
